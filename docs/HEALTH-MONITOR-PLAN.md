## 🔧 工具脚本

### heartbeat-memory-check.sh

```bash
#!/bin/bash
# 心跳记忆系统检查

MEMORY_DIR="/root/.openclaw/workspace/memory"
MEMORY_FILE="${MEMORY_DIR}/../MEMORY.md"
ARCHIVE_DIR="${MEMORY_DIR}/archive"

echo "🧠 记忆系统健康检查"
echo "===================="

# 检查 MEMORY.md
if [ -f "$MEMORY_FILE" ]; then
    tokens=$(wc -c < "$MEMORY_FILE")
    echo "MEMORY.md: $tokens 字符"

    if [ $tokens -gt 8000 ]; then
        echo "⚠️  警告: MEMORY.md 过大"
    fi
else
    echo "❌ 错误: MEMORY.md 不存在"
fi

# 检查日志数量
current_count=$(find "$MEMORY_DIR" -maxdepth 1 -name "*.md" | wc -l)
echo "当前日志: $current_count 个"

# 检查归档
archive_count=$(find "$ARCHIVE_DIR" -name "*.md" 2>/dev/null | wc -l)
echo "归档日志: $archive_count 个"

# 检查最近更新
latest_log=$(ls -t "$MEMORY_DIR"/*.md 2>/dev/null | head -1)
if [ -n "$latest_log" ]; then
    latest_date=$(stat -c %Y "$latest_log" 2>/dev/null || stat -f %m "$latest_log")
    current_time=$(date +%s)
    age_hours=$(( (current_time - latest_date) / 3600 ))
    echo "最新日志: ${age_hours} 小时前"

    if [ $age_hours -gt 48 ]; then
        echo "⚠️  警告: 超过 48 小时未更新日志"
    fi
fi

echo "===================="
echo "检查完成"
```
