## 🎯 设计模式 1: 安全升级模式

### 问题
- 升级时 Gateway 崩溃
- 配置不兼容
- 版本混乱

### 解决方案
```bash
safe_upgrade() {
    local new_version="$1"
    
    # 1. 备份当前状态
    echo "📦 备份配置..."
    backup_config "/root/.openclaw/backups/before-upgrade-$(date +%Y%m%d%H%M%S)"
    
    # 2. 检查兼容性
    echo "🔍 检查兼容性..."
    if ! check_compatibility "$new_version"; then
        echo "❌ 不兼容，取消升级"
        return 1
    fi
    
    # 3. 升级所有包管理器
    echo "⬆️ 升级 npm..."
    npm install -g "openclaw@$new_version"
    
    echo "⬆️ 升级 pnpm..."
    pnpm install -g "openclaw@$new_version"
    
    # 4. 验证版本
    echo "✅ 验证版本..."
    local npm_version=$(openclaw --version)
    local pnpm_version=$(pnpm list -g openclaw --depth=0 | grep openclaw | awk '{print $2}')
    
    if [[ "$npm_version" != "$new_version" ]] || [[ "$pnpm_version" != "$new_version" ]]; then
        echo "⚠️ 版本不一致，回滚"
        restore_backup
        return 1
    fi
    
    # 5. 测试功能
    echo "🧪 测试功能..."
    if ! openclaw doctor; then
        echo "❌ 测试失败，回滚"
        restore_backup
        return 1
    fi
    
    echo "✅ 升级成功！"
    return 0
}
```

### 使用方式
```bash
# 升级到新版本
safe_upgrade "2026.3.8"
```

## 🎯 设计模式 2: 安全文件读取模式

### 问题
- 读取不存在的文件
- 读取目录而不是文件
- 缺少错误处理

### 解决方案
```bash
# Bash 版本
safe_read_file() {
    local path="$1"
    local default="${2:-}"
    
    # 检查路径类型
    if [[ -d "$path" ]]; then
        echo "❌ 错误: $path 是目录，不是文件" >&2
        return 1
    elif [[ ! -e "$path" ]]; then
        if [[ -n "$default" ]]; then
            echo "$default"
            return 0
        else
            echo "❌ 错误: $path 不存在" >&2
            return 1
        fi
    fi
    
    # 读取文件
    cat "$path"
    return $?
}

# Python 版本
import os

def safe_read_file(path: str, default: str = None) -> str:
    """安全读取文件"""
    # 检查路径类型
    if os.path.isdir(path):
        raise ValueError(f"错误: {path} 是目录，不是文件")
    
    if not os.path.exists(path):
        if default is not None:
            return default
        else:
            raise FileNotFoundError(f"错误: {path} 不存在")
    
    # 读取文件
    with open(path, 'r') as f:
        return f.read()
```

### 使用方式
```bash
# 读取文件，提供默认值
content=$(safe_read_file "/path/to/file.txt" "默认内容")

# 检查返回值
if [[ $? -eq 0 ]]; then
    echo "读取成功: $content"
else
    echo "读取失败"
fi
```

## 🎯 设计模式 3: 安全命令执行模式

### 问题
- 命令超时
- 进程被终止
- 缺少资源监控

### 解决方案
```bash
safe_execute() {
    local timeout="${1:-300}"  # 默认 5 分钟
    shift
    local command="$@"
    
    # 执行命令，带超时
    timeout "$timeout" $command <<'EOF'
    echo "命令执行中..."
    eval "$command"
EOF
    
    local exit_code=$?
    
    if [[ $exit_code -eq 124 ]]; then
        echo "❌ 错误: 命令超时 (${timeout}s)" >&2
        return 124
    elif [[ $exit_code -eq 143 ]]; then
        echo "⚠️ 警告: 命令被终止" >&2
        return 143
    elif [[ $exit_code -ne 0 ]]; then
        echo "❌ 错误: 命令失败 (退出码: $exit_code)" >&2
        return $exit_code
    fi
    
    return 0
}

# 带资源监控的版本
safe_execute_with_monitor() {
    local timeout="$1"
    shift
    local command="$@"
    
    # 启动监控
    (
        while true; do
            local mem_usage=$(free | awk '/Mem/{printf "%.1f", $3/$2*100}')
            echo "内存使用: ${mem_usage}%"
            sleep 5
        done
    ) &
    local monitor_pid=$!
    
    # 执行命令
    safe_execute "$timeout" $command
    local exit_code=$?
    
    # 停止监控
    kill $monitor_pid 2>/dev/null
    
    return $exit_code
}
```

### 使用方式
```bash
# 执行命令，5分钟超时
safe_execute 300 python3 script.py

# 带内存监控
safe_execute_with_monitor 300 python3 script.py
```

## 🎯 设计模式 4: 版本管理模式

### 问题
- 文档不同步
- 版本命名混乱
- 多个版本分支

### 解决方案
```bash
# 版本更新脚本
update_version() {
    local new_version="$1"
    local update_files=(
        "SOUL.md"
        "README.md"
        "HEARTBEAT.md"
        "package.json"
    )
    
    # 1. 更新所有文件中的版本号
    for file in "${update_files[@]}"; do
        if [[ -f "$file" ]]; then
            sed -i "s/版本:.*/版本: $new_version/" "$file"
            echo "✅ 更新 $file"
        fi
    done
    
    # 2. 创建更新日志
    cat > "CHANGELOG.md" <<EOF
# 更新日志 - $new_version

**更新时间**: $(date +%Y-%m-%d)

## 新增
- 

## 修复
- 

## 改进
- 
EOF
    
    # 3. 删除旧版本文档
    find . -name "*v5.*" -o -name "*v6.*" | grep -v "$new_version" | xargs rm -v
    
    # 4. 验证一致性
    echo "🔍 验证版本一致性..."
    local versions=$(grep -r "版本:" ${update_files[@]} | awk '{print $2}' | sort -u)
    if [[ $(echo "$versions" | wc -l) -ne 1 ]]; then
        echo "⚠️ 警告: 版本号不一致"
        echo "$versions"
    else
        echo "✅ 版本号一致"
    fi
}
```

### 使用方式
```bash
# 更新到新版本
update_version "5.5"
```

## 🎯 设计模式 5: 错误处理模式

### 问题
- 错误信息不清晰
- 缺少上下文
- 没有恢复建议

### 解决方案
```bash
# 统一错误处理
handle_error() {
    local error_code="$1"
    local error_msg="$2"
    local context="${3:-无上下文}"
    local suggestion="${4:-无建议}"
    
    # 记录错误
    echo "❌ 错误: $error_msg" >&2
    echo "📍 位置: $context" >&2
    echo "💡 建议: $suggestion" >&2
    
    # 保存到错误日志
    local error_log="/root/.openclaw/workspace/logs/errors.log"
    {
        echo "=== 错误 $(date +%Y-%m-%d\ %H:%M:%S) ==="
        echo "错误码: $error_code"
        echo "错误信息: $error_msg"
        echo "上下文: $context"
        echo "建议: $suggestion"
        echo ""
    } >> "$error_log"
    
    # 返回错误码
    return "$error_code"
}

# 使用示例
if ! safe_read_file "/path/to/file"; then
    handle_error 1 "文件读取失败" "safe_read_file" "检查文件路径是否存在"
fi
```
